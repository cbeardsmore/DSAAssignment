    public DSALinkedList searchDate(String date)

 289

 291

 292 
 

 293

 297

 298 
        while ( ( iter.hasNext() ) && ( item.getWar().compareTo(maxDate) <= 0 ) )

 299

 300 
            //Add any matching values to a new linked list

 301 
            matches.insertFirst( item );

 302 
            item = (Carton)iter.next();

 303 
        }    

 304 
 

 305
